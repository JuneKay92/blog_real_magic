Example

### Need two files.  One for the code to run and one for setting up and running the tasks
# my_scheduler
the_time <- Sys.time()
print(the_time)
# task_setup
taskscheduler_create(taskname = "myfancyscript2", rscript = "C:/Users/hopesnyder/Desktop/scheduling/my_scheduler.R", 
                     schedule = "ONCE")
# task_run
taskscheduler_runnow(taskname = "testing")
