Example of a song

library(BRRR)
library(purrr)
library(tibble)

size <- 5

rap_beats <- function(size) {
  
  x <- data.frame(artist = sample(1:length(skrrrahh_list()), 
                                  size = size, 
                                  replace = TRUE), 
                  delay = sample(seq(from = .1, to = 1, by = .1), 
                                 size = size,
                                 replace = TRUE))
  
  rap_wait <- function(sound, sleep){
    skrrrahh(sound)
    Sys.sleep(sleep)
  }
  
  walk2(.x = x$artist, .y = x$delay, rap_wait)
}

rap_beats(50)
